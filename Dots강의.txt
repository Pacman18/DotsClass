---- Job System을 배워보자 ---------------

Job은 유니티에서 사용가능한 멀티쓰레드 API다. 

Dots의 ECS시스템과 별개라고 생각하면된다. 

단 연산가능한것이 값타입위주 그리고 NativeArray등의 컨테이너만 가능하다. 

new NativeArray<float>(count, Allocator.TYPE);
Allocator타입은 잠깐쓰는거 계속쓸꺼 등등이 있다. 

struct MyJob : IJob
{
	public int temn_1;
	public temp_2;

	public void Excute()
	{
		// temn_1 , temp2를 알아서 지지고 복고함. 	
	}
}


----------------------------- Hybrid 타입 알아보기 ------------------------------------------------

Mono월드에서 ECS월드 받아오기 
var world = World.DefaultGameObjectInjectionWorld;

EntityManager 받아오기 

var entityManager = world.EntityManager;

쿼리만들기 

var entityQuery = entityManager.CreateEntityQuery(typeof(T));

// 쿼리를  Mono에서 돌리기위한 변환 
var entities = entityQuery.ToEntityArray(Allocator.Temp);

foreach(var entity in entities)
{
     var santaCube = entityManager.GetComponentData<T>(entity);
     Debug.Log("SantaCube: " + santaCube.number);
}
